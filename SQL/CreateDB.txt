CREATE DATABASE QLSV
GO
USE QLSV
GO
CREATE TABLE Khoa(
MaKhoa NVARCHAR(5) NOT NULL,
TenKhoa NVARCHAR(20) NULL,
Phone NVARCHAR(10) NULL,
PRIMARY KEY(MaKhoa)
)
GO
CREATE TABLE SinhVien(
MaSV NVARCHAR(5) NOT NULL,
HoTen NVARCHAR(30) NULL,
DiaChi NVARCHAR(30) NULL,
NgaySinh DATE NULL,
LopBC NVARCHAR(10) NULL,
MaKhoa NVARCHAR(5) NOT NULL,
PRIMARY KEY(MaSV),
FOREIGN KEY(MaKhoa)
REFERENCES dbo.Khoa(MaKhoa),
)
GO
CREATE TABLE MonHoc(
MaMon NVARCHAR(5) NOT NULL,
TenMon NVARCHAR(20) NULL,
SoTinChi INT NULL,
GhiChu NVARCHAR(50) NULL,
MaKhoa NVARCHAR(5) NOT NULL,
PRIMARY KEY(MaMon),
FOREIGN KEY(MaKhoa)
REFERENCES dbo.Khoa(MaKhoa),
)
GO
CREATE TABLE DangKyHoc(
MaDangKy NVARCHAR(5) NOT NULL,
MaSV NVARCHAR(5) NOT NULL,
MaMon NVARCHAR(5) NOT NULL,
NamHK INT NULL,
GhiChu NVARCHAR(50) NULL,
PRIMARY KEY(MaDangKy),
FOREIGN KEY(MaSV)
REFERENCES dbo.SinhVien(MaSV),
FOREIGN KEY(MaMon)
REFERENCES dbo.MonHoc(MaMon),
)
